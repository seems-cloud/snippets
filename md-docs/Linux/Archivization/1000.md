```bash
zstd -3 -z -T16 -f -k file
lz4 -6 -T16 -z -f -k file
pigz -6 -p16 -k file
```

```bash
tar -cf - file | pv | pigz -6 -p16 > arch.tgz

tar -cf - file | pv | gzip -6 > arch.tgz
tar -cf - file | pv | bzip2 -6 > arch.tbz2
tar -cf - file | pv | xz -6 > arch.txz
tar -cf - file | pv | lz4 -6 -T16 > arch.tz4
```

```bash
tar -czf arch.tgz file
tar -cjf arch.tbz2 file
tar -cJf arch.txz file
tar -cf - file | lz4 -6 -T16 > arch.tz4
```
