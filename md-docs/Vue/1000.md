## Parent

```vue
<template>
  <OfferType
    :selectedType="selectedType"
    @selected-offer-type="updateSelectedOfferType"
  />
</template>

<script>

export default {
  // ...
  data() {
    return {
      selectedType: null
    };
  },
  methods: {
    updateSelectedType(type) {
      this.selectedType = type;
    },
  },
};
</script>
```

## Child

```vue
<template>
  <button @click="selectType('rent')"/>
</template>
  
<script>
export default {
  props: {
    selectedOfferType: {
      type: String,
      default: null,
    },
  },
  data() {
    return {
      localSelectedType: this.selectedType,
    };
  },
  methods: {
    selectType(type) {
      this.localSelectedOfferType = type;
      this.$emit("selected-offer-type", type);
    },
  },
};
</script>
```
